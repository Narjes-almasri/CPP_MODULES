	HitPoint = d.HitPoint;

	EnergyPoint = d.EnergyPoint;
}
ClapTrap &ClapTrap::operator=(const ClapTrap& d)
{
	std::cout<<"ClapTrap copy assignment constructor called\n";
	if(this != &d)
	{
		Name = d.Name;
		HitPoint = d.HitPoint;
		AttackDam = d.AttackDam;
		EnergyPoint = d.EnergyPoint;
	}
	return (*this);
}

{
	std::cout<<"ClapTrap Destructor called\n";
}
